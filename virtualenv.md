
  # Make account:

  <code>
        sudo adduser $yourname
  
        #input your information
  </code>

  # How to install virtualenv:

  <code>
  
      #log in with your account
      virtualenv -p python3 $yourname 
      
  </code>

  # Active your virtual environment:
  
  <code>
  
    source $yourname/bin/activate
    
  </code>

  # To deactivate:
  
  <code>
  
    deactivate
    
  </code>
